class SingletonMeta(type):
    _instances = {}

    # –ö–æ–Ω—Å—Ç—Ä—É–∫—Ç–æ—Ä (—Å–æ–∑–¥–∞–Ω–∏–µ —Ä–æ–±–æ—Ç–∞)

## –ù–∞—Å–ª–µ–¥–æ–≤–∞–Ω–∏–µ (Inheritance) üë®‚Äçüë©‚Äçüëß‚Äçüë¶

### –ß—Ç–æ —ç—Ç–æ:

    def eat(self):  # –ü–µ—Ä–µ–æ–ø—Ä–µ–¥–µ–ª—è–µ–º –º–µ—Ç–æ–¥ —Ä–æ–¥–∏—Ç–µ–ª—è
        print(f"{self.name} –µ—Å—Ç –∫–æ—Ä–º")

# –°–æ–∑–¥–∞—ë–º –æ–±—ä–µ–∫—Ç—ã

from abc import ABC, abstractmethod

class Shape(ABC):  # –ê–±—Å—Ç—Ä–∞–∫—Ç–Ω—ã–π –∫–ª–∞—Å—Å
    def __init__(self, name):
        self.name = name
    
    @abstractmethod
    def area(self):
        pass  # –î–æ–ª–∂–µ–Ω –±—ã—Ç—å —Ä–µ–∞–ª–∏–∑–æ–≤–∞–Ω –≤ –¥–æ—á–µ—Ä–Ω–µ–º –∫–ª–∞—Å—Å–µ
    
    @abstractmethod
    def perimeter(self):
        pass  # –î–æ–ª–∂–µ–Ω –±—ã—Ç—å —Ä–µ–∞–ª–∏–∑–æ–≤–∞–Ω –≤ –¥–æ—á–µ—Ä–Ω–µ–º –∫–ª–∞—Å—Å–µ
    
    def describe(self):
        print(f"–≠—Ç–æ {self.name}")

        super().__init__("–ø—Ä—è–º–æ—É–≥–æ–ª—å–Ω–∏–∫")
        self.width = width
        self.height = height
    
    def area(self):
        return self.width * self.height
    
    def perimeter(self):
        return 2 * (self.width + self.height)

        super().__init__("–∫—Ä—É–≥")
        self.radius = radius
    
    def area(self):
        return 3.14159 * self.radius ** 2
    
    def perimeter(self):
        return 2 * 3.14159 * self.radius

    Circle(4)

    print(f"–ü–µ—Ä–∏–º–µ—Ç—Ä: {shape.perimeter()}")
